	var arr=[2,4,2,54,7,5,11];
	console.log(arr);//(7) [2, 4, 2, 54, 7, 5, 11]
	var n=arr.forEach(function(item,index){
		console.log(index+':'+item);
		// 无法return
	});
		// 0:2
		// 1:4
		// 2:2
		// 3:54
		// 4:7
		// 5:5
		// 6:11

数组filter遍历
过滤 查找  原数组不改变
```
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>Document</title>
</head>
<body>
	
</body>
<script>
	
	var arr=[2,4,2,54,7,5,11];
	console.log(arr);//(7) [2, 4, 2, 54, 7, 5, 11]
	var n=arr.filter(function(item,index){
		if(item===4||item===2) return item;
	});
	console.log(n);//(3) [2, 4, 2]
</script>
</html>
```
